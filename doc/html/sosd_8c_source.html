184 void daemon_init() {
185  pid_t pid, sid;
186 
187  /* [fork]
188  * split off from the parent process (& terminate parent)
189  */
190  pid = fork();
191  if (pid < 0) {
192  fprintf(stderr, "Unable to start daemon (%s): Could not fork() off parent process.\n", DAEMON_NAME);
193  exit(EXIT_FAILURE);
194  }
195  if (pid > 0) { exit(EXIT_SUCCESS); } //close the parent
196 
197  sprintf(daemon_pid_str, "%d", getpid());
198  fprintf(stderr, "Starting daemon (%s) with PID = %s\n", DAEMON_NAME, daemon_pid_str);
199 
200  /* [child session]
201  * create/occupy independent session from parent process
202  */
203  umask(0);
204  sid = setsid();
205  if (sid < 0) {
206  fprintf(stderr, "Unable to start daemon (%s): Could not acquire a session id.\n", DAEMON_NAME);
207  exit(EXIT_FAILURE);
208  }
209  if ((chdir(WORK_DIR)) < 0) {
210  fprintf(stderr, "Unable to start daemon (%s): Could not change to working directory: %s\n", \
211  DAEMON_NAME, WORK_DIR);
212  exit(EXIT_FAILURE);
213  }
214 
215  /* [lock file]
216  * create and hold lock file to prevent multiple daemon spawn
217  */
218  lock_fptr = open(LOCK_FILE, O_RDWR | O_CREAT, 0640);
219  if (lock_fptr < 0) {
220  fprintf(stderr, "\nUnable to start daemon (%s): Could not access lock file %s in directory %s\n", \
221  DAEMON_NAME, LOCK_FILE, WORK_DIR);
222  exit(EXIT_FAILURE);
223  }
224  if (lockf(lock_fptr, F_TLOCK, 0) < 0) {
225  fprintf(stderr, "\nUnable to start daemon (%s): An instance is already running.\n", DAEMON_NAME);
226  exit(EXIT_FAILURE);
227  }
228  write(lock_fptr, daemon_pid_str, strlen(daemon_pid_str));

254  switch (signal) {
255 
256  case SIGHUP:
257  /* Attempt to refresh network connections
258  * NOTE: some systems use SIGHUP to reset servers...
259  */
260  syslog(LOG_DEBUG, "SIGHUP signal caught.");

262

264  /* Future-proofing for non-blocking socket accept calls... */
265  daemon_running = 0;
